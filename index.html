-- ═══════════════════════════════════════════
-- جدول الحملات التسويقية
-- ═══════════════════════════════════════════
CREATE TABLE campaigns (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255) NOT NULL,
    type ENUM('email','social','ppc','seo','content') NOT NULL,
    budget DECIMAL(10,2) NOT NULL,
    start_date DATE NOT NULL,
    end_date DATE,
    status ENUM('active','paused','completed') DEFAULT 'active',
    target_audience TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ═══════════════════════════════════════════
-- جدول مقاييس الأداء
-- ═══════════════════════════════════════════
CREATE TABLE campaign_metrics (
    id INT PRIMARY KEY AUTO_INCREMENT,
    campaign_id INT,
    impressions INT DEFAULT 0,
    clicks INT DEFAULT 0,
    conversions INT DEFAULT 0,
    revenue DECIMAL(12,2) DEFAULT 0,
    cost DECIMAL(12,2) DEFAULT 0,
    date DATE NOT NULL,
    FOREIGN KEY (campaign_id) REFERENCES campaigns(id) ON DELETE CASCADE
);

-- ═══════════════════════════════════════════
-- جدول العملاء
-- ═══════════════════════════════════════════
CREATE TABLE customers (
    id INT PRIMARY KEY AUTO_INCREMENT,
    email VARCHAR(255) UNIQUE NOT NULL,
    name VARCHAR(255),
    phone VARCHAR(20),
    source VARCHAR(100),
    segment ENUM('new','returning','vip') DEFAULT 'new',
    lifetime_value DECIMAL(12,2) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ═══════════════════════════════════════════
-- جدول السلات المهجورة
-- ═══════════════════════════════════════════
CREATE TABLE abandoned_carts (
    id INT PRIMARY KEY AUTO_INCREMENT,
    customer_id INT,
    cart_value DECIMAL(10,2) NOT NULL,
    items TEXT,
    abandoned_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    reminder_sent BOOLEAN DEFAULT FALSE,
    FOREIGN KEY (customer_id) REFERENCES customers(id)
);

-- ═══════════════════════════════════════════
-- جدول الزوار
-- ═══════════════════════════════════════════
CREATE TABLE visitors (
    id INT PRIMARY KEY AUTO_INCREMENT,
    ip_address VARCHAR(45),
    user_agent TEXT,
    source VARCHAR(100),
    page_visited VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ═══════════════════════════════════════════
-- جدول حملات الإيميل
-- ═══════════════════════════════════════════
CREATE TABLE email_campaigns (
    id INT PRIMARY KEY AUTO_INCREMENT,
    campaign_id INT,
    subject VARCHAR(255) NOT NULL,
    body TEXT NOT NULL,
    sent_count INT DEFAULT 0,
    open_count INT DEFAULT 0,
    click_count INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (campaign_id) REFERENCES campaigns(id)
);

// server.js - خادم التطبيق
// ═══════════════════════════════════════════
const express = require('express');
const mysql = require('mysql2/promise');
const cors = require('cors');
const jwt = require('jsonwebtoken');
require('dotenv').config();

const app = express();
app.use(cors());
app.use(express.json());

// اتصال قاعدة البيانات
const pool = mysql.createPool({
    host: process.env.DB_HOST || 'localhost',
    user: process.env.DB_USER || 'root',
    password: process.env.DB_PASSWORD || 'password',
    database: process.env.DB_NAME || 'marketing_db',
    waitForConnections: true,
    connectionLimit: 10
});

// Middleware التحقق
const auth = (req, res, next) => {
    const token = req.headers.authorization?.split(' ')[1];
    if (!token) return res.status(401).json({ error: 'Access denied' });
    jwt.verify(token, process.env.JWT_SECRET, (err, user) => {
        if (err) return res.status(403).json({ error: 'Invalid token' });
        req.user = user;
        next();
    });
};

// ═══════════════════════════════════════════
// 📊 إحصائيات لوحة التحكم
// ═══════════════════════════════════════════
app.get('/api/dashboard/stats', auth, async (req, res) => {
    try {
        const [visitors] = await pool.query(
            `SELECT COUNT(*) as total FROM visitors 
             WHERE created_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)`
        );
        const [revenue] = await pool.query(
            `SELECT COALESCE(SUM(revenue), 0) as total 
             FROM campaign_metrics 
             WHERE date >= DATE_SUB(NOW(), INTERVAL 30 DAY)`
        );
        const [conversion] = await pool.query(
            `SELECT COALESCE(SUM(clicks), 0) as clicks,
                    COALESCE(SUM(conversions), 0) as conv
             FROM campaign_metrics 
             WHERE date >= DATE_SUB(NOW(), INTERVAL 30 DAY)`
        );
        const cvr = conversion[0].clicks > 0 
            ? (conversion[0].conv / conversion[0].clicks * 100).toFixed(2) 
            : 0;

        res.json({
            visitors: visitors[0].total,
            revenue: revenue[0].total,
            conversionRate: parseFloat(cvr)
        });
    } catch (e) { res.status(500).json({ error: e.message }); }
});

// ═══════════════════════════════════════════
// 📢 إدارة الحملات
// ═══════════════════════════════════════════
app.post('/api/campaigns', auth, async (req, res) => {
    const { name, type, budget, start_date, target_audience } = req.body;
    try {
        const [result] = await pool.query(
            `INSERT INTO campaigns (name, type, budget, start_date, target_audience) 
             VALUES (?, ?, ?, ?, ?)`,
            [name, type, budget, start_date, target_audience]
        );
        res.status(201).json({ id: result.insertId, message: 'Created' });
    } catch (e) { res.status(500).json({ error: e.message }); }
});

app.get('/api/campaigns', auth, async (req, res) => {
    try {
        const [rows] = await pool.query(
            `SELECT c.*, 
                COALESCE(SUM(m.revenue), 0) as total_revenue,
                COALESCE(SUM(m.clicks), 0) as total_clicks,
                COALESCE(SUM(m.conversions), 0) as total_conversions
             FROM campaigns c
             LEFT JOIN campaign_metrics m ON c.id = m.campaign_id
             GROUP BY c.id ORDER BY c.created_at DESC`
        );
        res.json(rows);
    } catch (e) { res.status(500).json({ error: e.message }); }
});

app.patch('/api/campaigns/:id/status', auth, async (req, res) => {
    try {
        await pool.query('UPDATE campaigns SET status = ? WHERE id = ?',
            [req.body.status, req.params.id]);
        res.json({ message: 'Updated' });
    } catch (e) { res.status(500).json({ error: e.message }); }
});

app.delete('/api/campaigns/:id', auth, async (req, res) => {
    try {
        await pool.query('DELETE FROM campaign_metrics WHERE campaign_id = ?', [req.params.id]);
        await pool.query('DELETE FROM campaigns WHERE id = ?', [req.params.id]);
        res.json({ message: 'Deleted' });
    } catch (e) { res.status(500).json({ error: e.message }); }
});

// ═══════════════════════════════════════════
// 👥 العملاء
// ═══════════════════════════════════════════
app.post('/api/customers', auth, async (req, res) => {
    const { email, name, phone, source } = req.body;
    try {
        const [r] = await pool.query(
            'INSERT INTO customers (email, name, phone, source) VALUES (?, ?, ?, ?)',
            [email, name, phone, source]
        );
        res.status(201).json({ id: r.insertId });
    } catch (e) {
        if (e.code === 'ER_DUP_ENTRY') res.status(409).json({ error: 'Email exists' });
        else res.status(500).json({ error: e.message });
    }
});

// ═══════════════════════════════════════════
// 🔐 تسجيل الدخول
// ═══════════════════════════════════════════
app.post('/api/auth/login', async (req, res) => {
    const { email, password } = req.body;
    if (email === 'admin@marketing.com' && password === 'admin123') {

        );
        res.json({ token, user: { id: 1, email, name: 'Admin' } });
    } else {
        res.status(401).json({ error: 'Invalid credentials' });
    }
});

const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log(`🚀 Server on port ${PORT}`));

#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""بوت التسويق الذكي - Automated Marketing Bot"""

import schedule
import time
import smtplib
import os
import logging
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from datetime import datetime
import mysql.connector

logging.basicConfig(level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s')
logger = logging.getLogger(__name__)

class MarketingBot:
    def __init__(self):
        self.db = mysql.connector.connect(
            host=os.getenv('DB_HOST', 'localhost'),
            user=os.getenv('DB_USER', 'root'),
            password=os.getenv('DB_PASSWORD', 'password'),
            database=os.getenv('DB_NAME', 'marketing_db')
        )
        self.smtp = {
            'server': os.getenv('SMTP_SERVER', 'smtp.gmail.com'),
            'port': int(os.getenv('SMTP_PORT', '587')),
            'user': os.getenv('SMTP_USER'),
            'pass': os.getenv('SMTP_PASS')
        }

    def send_email(self, to, subject, body, html=None):
        """إرسال بريد إلكتروني"""
        try:
            msg = MIMEMultipart('alternative')
            msg['From'] = self.smtp['user']
            msg['To'] = to
            msg['Subject'] = subject
            msg.attach(MIMEText(body, 'plain', 'utf-8'))
            if html:
                msg.attach(MIMEText(html, 'html', 'utf-8'))

            with smtplib.SMTP(self.smtp['server'], self.smtp['port']) as s:
                s.starttls()
                s.login(self.smtp['user'], self.smtp['pass'])
                s.send_message(msg)
            logger.info(f"📧 Sent to {to}")
            return True
        except Exception as e:
            logger.error(f"❌ Email failed: {e}")
            return False

    def abandoned_carts(self):
        """معالجة السلات المهجورة"""
        logger.info("🛒 Processing abandoned carts...")
        cursor = self.db.cursor(dictionary=True)
        cursor.execute("""
            SELECT c.email, c.name, ac.cart_value, ac.items, ac.id
            FROM abandoned_carts ac
            JOIN customers c ON ac.customer_id = c.id
            WHERE ac.reminder_sent = FALSE 
            AND ac.abandoned_at < DATE_SUB(NOW(), INTERVAL 1 HOUR)
        """)

        for cart in cursor.fetchall():
            discount = "15" if cart['cart_value'] > 100 else "10"
            subject = f"🛒 {cart['name']}, you left items in your cart!"
            body = f"""Hi {cart['name']},

You left ${cart['cart_value']} worth of items in your cart.

Use code SAVE{discount} for {discount}% off!
Valid for 24 hours only.

[Complete Order]"""

            html = f"""
                
🛒 You forgot something!

                
Hi {cart['name']},

                
You left ${cart['cart_value']} in your cart.

                

                    
🎁 {discount}% OFF

                    
Code: SAVE{discount}

                

                
Complete Order →

            """

            if self.send_email(cart['email'], subject, body, html):
                cursor.execute("UPDATE abandoned_carts SET reminder_sent=TRUE WHERE id=%s", (cart['id'],))
                self.db.commit()
            time.sleep(2)
        cursor.close()

    def weekly_report(self):
        """تقرير الأداء الأسبوعي"""
        logger.info("📊 Generating weekly report...")
        cursor = self.db.cursor(dictionary=True)
        cursor.execute("""
            SELECT c.name, c.type, c.budget,
                   COALESCE(SUM(m.revenue),0) as rev,
                   COALESCE(SUM(m.clicks),0) as clicks,
                   COALESCE(SUM(m.conversions),0) as conv
            FROM campaigns c
            LEFT JOIN campaign_metrics m ON c.id=m.campaign_id
            WHERE m.date >= DATE_SUB(NOW(), INTERVAL 7 DAY)
            GROUP BY c.id ORDER BY rev DESC
        """)

        report = "📊 Weekly Marketing Report\n" + "="*50 + "\n\n"
        for camp in cursor.fetchall():
            roi = ((camp['rev']-camp['budget'])/camp['budget']*100) if camp['budget']>0 else 0
            report += f"📢 {camp['name']} ({camp['type']})\n"
            report += f"   Budget: ${camp['budget']} | Revenue: ${camp['rev']} | ROI: {roi:.1f}%\n"
            report += f"   Clicks: {camp['clicks']} | Conversions: {camp['conv']}\n\n"
        cursor.close()

        self.send_email(os.getenv('REPORT_EMAIL', 'manager@company.com'),
            'Weekly Marketing Report', report)

    def update_segments(self):
        """تحديث شرائح العملاء"""
        logger.info("🔄 Updating customer segments...")
        cursor = self.db.cursor()
        cursor.execute("UPDATE customers SET segment='vip' WHERE lifetime_value > 5000")
        cursor.execute("""UPDATE customers SET segment='returning' 
            WHERE lifetime_value > 500 AND lifetime_value <= 5000 
            AND created_at < DATE_SUB(NOW(), INTERVAL 3 MONTH)""")
        self.db.commit()
        cursor.close()

    def run(self):
        """تشغيل البوت"""
        logger.info("🤖 Marketing Bot started!")
        schedule.every().hour.do(self.abandoned_carts)
        schedule.every().sunday.at("09:00").do(self.weekly_report)
        schedule.every(2).days.at("02:00").do(self.update_segments)

        while True:
            schedule.run_pending()
            time.sleep(60)

if __name__ == "__main__":
    bot = MarketingBot()
    bot.run()
